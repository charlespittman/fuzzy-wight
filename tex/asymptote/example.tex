\begin{figure}[]
  \centering
  \begin{asy}
    size(8cm);
    include graph;
    Label f;
    f.p = fontsize(6);
    xaxis(-10,10, Ticks(f, 2.0, 0.5));
    yaxis(-10,10, Ticks(f, 2.0));

    real f(real x)
    {
      return (x^3);
    }
    draw(graph(f, -2, 2));
  \end{asy}
\end{figure}
